4-2. emp 테이블에서 각 부서별 사원수 조회
[3]
SELECT deptno
        ,COUNT(*)
        ,SUM( sal+NVL(comm,0)) 
        ,AVG( sal+NVL(comm,0)) 
        ,MAX( sal+NVL(comm,0)) 
        ,MIN( sal+NVL(comm,0)) 
FROM emp
GROUP BY deptno     -- GROUP BY 는 집계함수를 반드시사용
ORDER BY deptno;
